# rift Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build rift — a single Go binary (`server`/`client` subcommands) that tunnels localhost services to the internet over QUIC, with HTTP subdomain routing and raw TCP forwarding.

**Architecture:** QUIC server on port 443 authenticates clients via token, assigns HTTP subdomains or TCP ports, routes visitor traffic by opening QUIC data streams to the registered client, which relays to the local service. Visitor TLS is Let's Encrypt per-subdomain (prod) or self-signed wildcard (dev). UDP 443 = QUIC; TCP 443 = HTTPS — no conflict.

**Tech Stack:** Go 1.22+, `github.com/quic-go/quic-go`, `golang.org/x/crypto/acme/autocert`, `log/slog`, standard library only.

package proto

import (
	"encoding/binary"
	"encoding/json"
	"fmt"
	"io"
)

const maxMsgSize = 1 << 20 // 1 MiB

// WriteMsg encodes msg as JSON and writes it with a 4-byte big-endian length prefix.
func WriteMsg(w io.Writer, msg *ControlMsg) error {
	data, err := json.Marshal(msg)
	if err != nil {
		return fmt.Errorf("marshal control msg: %w", err)
	}
	if len(data) > maxMsgSize {
		return fmt.Errorf("message too large: %d bytes", len(data))
	}
	var hdr [4]byte
	binary.BigEndian.PutUint32(hdr[:], uint32(len(data)))
	if _, err := w.Write(hdr[:]); err != nil {
		return fmt.Errorf("write length prefix: %w", err)
	}
	if _, err := w.Write(data); err != nil {
		return fmt.Errorf("write payload: %w", err)
	}
	return nil
}

// ReadMsg reads a length-prefixed JSON message from r.
func ReadMsg(r io.Reader) (*ControlMsg, error) {
	var hdr [4]byte
	if _, err := io.ReadFull(r, hdr[:]); err != nil {
		return nil, fmt.Errorf("read length prefix: %w", err)
	}
	n := binary.BigEndian.Uint32(hdr[:])
	if n > maxMsgSize {
		return nil, fmt.Errorf("message too large: %d bytes", n)
	}
	buf := make([]byte, n)
	if _, err := io.ReadFull(r, buf); err != nil {
		return nil, fmt.Errorf("read payload: %w", err)
	}
	var msg ControlMsg
	if err := json.Unmarshal(buf, &msg); err != nil {
		return nil, fmt.Errorf("unmarshal control msg: %w", err)
	}
	return &msg, nil
}
```

package proto

import (
	"encoding/binary"
	"fmt"
	"io"
)

// TunnelHeader is the 8-byte prefix on every server-initiated data stream.
// Layout: [4 bytes tunnel_id big-endian][4 bytes reserved zero].
type TunnelHeader struct {
	TunnelID uint32
}

// WriteHeader writes the 8-byte tunnel header to w.
func WriteHeader(w io.Writer, h TunnelHeader) error {
	var buf [8]byte
	binary.BigEndian.PutUint32(buf[:4], h.TunnelID)
	if _, err := w.Write(buf[:]); err != nil {
		return fmt.Errorf("write tunnel header: %w", err)
	}
	return nil
}

// ReadHeader reads the 8-byte tunnel header from r.
func ReadHeader(r io.Reader) (TunnelHeader, error) {
	var buf [8]byte
	if _, err := io.ReadFull(r, buf[:]); err != nil {
		return TunnelHeader{}, fmt.Errorf("read tunnel header: %w", err)
	}
	return TunnelHeader{TunnelID: binary.BigEndian.Uint32(buf[:4])}, nil
}
```

package auth

import (
	"crypto/rand"
	"encoding/hex"
	"fmt"
	"strings"
)

// GenerateToken creates "t_" + 32 random bytes hex-encoded (66 chars total).
func GenerateToken() (string, error) {
	b := make([]byte, 32)
	if _, err := rand.Read(b); err != nil {
		return "", fmt.Errorf("generate token bytes: %w", err)
	}
	return "t_" + hex.EncodeToString(b), nil
}

// ValidateTokenFormat returns true if t is a well-formed rift token.
func ValidateTokenFormat(t string) bool {
	if !strings.HasPrefix(t, "t_") {
		return false
	}
	rest := t[2:]
	if len(rest) != 64 {
		return false
	}
	_, err := hex.DecodeString(rest)
	return err == nil
}

package auth

import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

// ServerTokenStore manages /etc/rift/tokens.
// Format per line: "name token" (# lines are comments).
type ServerTokenStore struct{ path string }

func NewServerTokenStore(path string) *ServerTokenStore { return &ServerTokenStore{path: path} }

// Validate returns true if token exists in the store.
func (s *ServerTokenStore) Validate(token string) (bool, error) {
	f, err := os.Open(s.path)
	if err != nil {
		return false, fmt.Errorf("open server token store: %w", err)
	}
	defer f.Close()
	sc := bufio.NewScanner(f)
	for sc.Scan() {
		line := strings.TrimSpace(sc.Text())
		if line == "" || strings.HasPrefix(line, "#") {
			continue
		}
		if parts := strings.Fields(line); len(parts) == 2 && parts[1] == token {
			return true, nil
		}
	}
	return false, sc.Err()
}

// Add appends "name token\n" to the store (creates it if absent, mode 0600).
func (s *ServerTokenStore) Add(name, token string) error {
	f, err := os.OpenFile(s.path, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0o600)
	if err != nil {
		return fmt.Errorf("open server token store for writing: %w", err)
	}
	defer f.Close()
	_, err = fmt.Fprintf(f, "%s %s\n", name, token)
	return err
}

// ClientTokenStore manages ~/.config/rift/tokens.
// Format per line: "server token".
type ClientTokenStore struct{ path string }

func NewClientTokenStore(path string) *ClientTokenStore { return &ClientTokenStore{path: path} }

// Lookup returns the token for the given server address.
func (c *ClientTokenStore) Lookup(server string) (string, error) {
	f, err := os.Open(c.path)
	if err != nil {
		return "", fmt.Errorf("open client token store: %w", err)
	}
	defer f.Close()
	sc := bufio.NewScanner(f)
	for sc.Scan() {
		line := strings.TrimSpace(sc.Text())
		if line == "" || strings.HasPrefix(line, "#") {
			continue
		}
		if parts := strings.Fields(line); len(parts) == 2 && parts[0] == server {
			return parts[1], nil
		}
	}
	if err := sc.Err(); err != nil {
		return "", err
	}
	return "", fmt.Errorf("no token found for server %q", server)
}

// Save appends "server token\n" to the client store.
func (c *ClientTokenStore) Save(server, token string) error {
	f, err := os.OpenFile(c.path, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0o600)
	if err != nil {
		return fmt.Errorf("open client token store for writing: %w", err)
	}
	defer f.Close()
	_, err = fmt.Fprintf(f, "%s %s\n", server, token)
	return err
}

package relay

import (
	"context"
	"io"
	"sync"
)

var bufPool = sync.Pool{
	New: func() any { return make([]byte, 32*1024) },
}

// Relay bidirectionally copies between dst and src until either side closes.
// Both directions run concurrently; Relay returns when both finish.
func Relay(ctx context.Context, dst io.ReadWriteCloser, src io.ReadWriteCloser) {
	var wg sync.WaitGroup
	wg.Add(2)
	go func() {
		defer wg.Done()
		copyBuf(dst, src)
		dst.Close()
	}()
	go func() {
		defer wg.Done()
		copyBuf(src, dst)
		src.Close()
	}()
	wg.Wait()
}

func copyBuf(dst io.Writer, src io.Reader) {
	buf := bufPool.Get().([]byte)
	defer bufPool.Put(buf)
	io.CopyBuffer(dst, src, buf)
}

// DevTLSConfig returns a TLS config with a self-signed wildcard cert for domain.
// Covers both "domain" and "*.domain". No ACME needed.
func DevTLSConfig(domain string) (*tls.Config, error) {
	key, err := ecdsa.GenerateKey(elliptic.P256(), rand.Reader)
	if err != nil {
		return nil, fmt.Errorf("generate dev key: %w", err)
	}
	tmpl := &x509.Certificate{
		SerialNumber: big.NewInt(1),
		Subject:      pkix.Name{CommonName: domain},
		DNSNames:     []string{domain, "*." + domain},
		NotBefore:    time.Now().Add(-time.Hour),
		NotAfter:     time.Now().Add(365 * 24 * time.Hour),
		KeyUsage:     x509.KeyUsageDigitalSignature,
		ExtKeyUsage:  []x509.ExtKeyUsage{x509.ExtKeyUsageServerAuth},
	}
	der, err := x509.CreateCertificate(rand.Reader, tmpl, tmpl, &key.PublicKey, key)
	if err != nil {
		return nil, fmt.Errorf("create self-signed cert: %w", err)
	}
	certPEM := pem.EncodeToMemory(&pem.Block{Type: "CERTIFICATE", Bytes: der})
	keyDER, err := x509.MarshalECPrivateKey(key)
	if err != nil {
		return nil, fmt.Errorf("marshal EC key: %w", err)
	}
	keyPEM := pem.EncodeToMemory(&pem.Block{Type: "EC PRIVATE KEY", Bytes: keyDER})
	cert, err := tls.X509KeyPair(certPEM, keyPEM)
	if err != nil {
		return nil, fmt.Errorf("load key pair: %w", err)
	}
	return &tls.Config{
		Certificates: []tls.Certificate{cert},
		NextProtos:   []string{"rift-v1", "h2", "http/1.1"},
	}, nil
}

// ProdTLSConfig returns a TLS config backed by Let's Encrypt autocert.
// Individual subdomain certs are obtained on first connection via TLS-ALPN-01.
// cacheDir stores ACME state (e.g. /etc/rift/certs).
func ProdTLSConfig(domain, cacheDir string) *tls.Config {
	m := &autocert.Manager{
		Cache:      autocert.DirCache(cacheDir),
		Prompt:     autocert.AcceptTOS,
		HostPolicy: autocert.HostWhitelist(domain, "*."+domain),
	}
	cfg := m.TLSConfig()
	// Prepend QUIC ALPN so the same TLS config works for both listeners.
	cfg.NextProtos = append([]string{"rift-v1"}, cfg.NextProtos...)
	return cfg
}

package server

import (
	"fmt"
	"math/rand/v2"
	"sync"
	"sync/atomic"
)

// Tunnel represents one active tunnel registration.
type Tunnel struct {
	ID        uint32
	Proto     string // "http" or "tcp"
	Subdomain string // http tunnels only
	TCPPort   uint16 // tcp tunnels only
	Conn      any    // *quic.Connection set at runtime; any to avoid import cycle in tests
}

// TunnelRegistry is a thread-safe store of active tunnels.
type TunnelRegistry struct {
	mu          sync.RWMutex
	bySubdomain map[string]*Tunnel
	byTCPPort   map[uint16]*Tunnel
	byID        map[uint32]*Tunnel
	nextID      atomic.Uint32
}

func NewTunnelRegistry() *TunnelRegistry {
	return &TunnelRegistry{
		bySubdomain: make(map[string]*Tunnel),
		byTCPPort:   make(map[uint16]*Tunnel),
		byID:        make(map[uint32]*Tunnel),
	}
}

// RegisterHTTP assigns a subdomain and returns the tunnel ID.
// If name is empty, a random 8-hex-char name is used.
func (r *TunnelRegistry) RegisterHTTP(name string, conn any) (uint32, error) {
	r.mu.Lock()
	defer r.mu.Unlock()
	if name == "" {
		name = randomHex(4)
	}
	if _, exists := r.bySubdomain[name]; exists {
		return 0, fmt.Errorf("subdomain %q already in use", name)
	}
	id := r.nextID.Add(1)
	t := &Tunnel{ID: id, Proto: "http", Subdomain: name, Conn: conn}
	r.bySubdomain[name] = t
	r.byID[id] = t
	return id, nil
}

// RegisterTCP assigns a random port in [10000, 65535] and returns (id, port, err).
func (r *TunnelRegistry) RegisterTCP(conn any) (uint32, uint16, error) {
	r.mu.Lock()
	defer r.mu.Unlock()
	port, err := r.allocPort()
	if err != nil {
		return 0, 0, err
	}
	id := r.nextID.Add(1)
	t := &Tunnel{ID: id, Proto: "tcp", TCPPort: port, Conn: conn}
	r.byTCPPort[port] = t
	r.byID[id] = t
	return id, port, nil
}

// Unregister removes the tunnel with the given ID from all indexes.
func (r *TunnelRegistry) Unregister(id uint32) {
	r.mu.Lock()
	defer r.mu.Unlock()
	t, ok := r.byID[id]
	if !ok {
		return
	}
	delete(r.byID, id)
	delete(r.bySubdomain, t.Subdomain)
	delete(r.byTCPPort, t.TCPPort)
}

func (r *TunnelRegistry) LookupBySubdomain(sub string) *Tunnel {
	r.mu.RLock()
	defer r.mu.RUnlock()
	return r.bySubdomain[sub]
}

func (r *TunnelRegistry) LookupByTCPPort(port uint16) *Tunnel {
	r.mu.RLock()
	defer r.mu.RUnlock()
	return r.byTCPPort[port]
}

func (r *TunnelRegistry) LookupByID(id uint32) *Tunnel {
	r.mu.RLock()
	defer r.mu.RUnlock()
	return r.byID[id]
}

// allocPort picks a random free port in [10000, 65535]. Caller must hold mu.
func (r *TunnelRegistry) allocPort() (uint16, error) {
	for range 100 {
		p := uint16(10000 + rand.IntN(55535))
		if _, used := r.byTCPPort[p]; !used {
			return p, nil
		}
	}
	return 0, fmt.Errorf("unable to allocate a free TCP port after 100 attempts")
}

func randomHex(n int) string {
	b := make([]byte, n)
	rand.Read(b) //nolint:errcheck // math/rand/v2 never errors
	return fmt.Sprintf("%x", b)
}

	"context"

	"github.com/user/rift/internal/auth"

// Config holds all server runtime configuration.
type Config struct {
	ListenAddr string
	Domain     string
	Dev        bool // skip token validation in dev mode
	TLS        *tls.Config
	TokenStore *auth.ServerTokenStore
	Registry   *TunnelRegistry
}

var quicConf = &quic.Config{
	MaxIdleTimeout:    30 * time.Second,
	KeepAlivePeriod:   15 * time.Second,
	MaxIncomingStreams: 1000,
	Allow0RTT:         true,
}

// Run starts the QUIC listener and blocks until ctx is cancelled.
func Run(ctx context.Context, cfg Config) error {
	tlsCfg := cfg.TLS.Clone()
	tlsCfg.NextProtos = []string{"rift-v1"}

	ln, err := quic.ListenAddr(cfg.ListenAddr, tlsCfg, quicConf)
	if err != nil {
		return fmt.Errorf("quic listen %s: %w", cfg.ListenAddr, err)
	}
	slog.Info("QUIC listener started", "addr", cfg.ListenAddr)
	defer ln.Close()

	go func() {
		<-ctx.Done()
		ln.Close()
	}()

	for {
		conn, err := ln.Accept(ctx)
		if err != nil {
			if ctx.Err() != nil {
				return nil
			}
			slog.Error("accept QUIC connection", "err", err)
			continue
		}
		go handleConn(ctx, conn, cfg)
	}
}

func handleConn(ctx context.Context, conn quic.Connection, cfg Config) {
	remote := conn.RemoteAddr().String()
	slog.Info("client connected", "remote", remote)
	defer func() {
		conn.CloseWithError(0, "bye")
		slog.Info("client disconnected", "remote", remote)
	}()

	// Client opens the first stream as the control stream.
	stream, err := conn.AcceptStream(ctx)
	if err != nil {
		slog.Error("accept control stream", "remote", remote, "err", err)
		return
	}

	// Auth handshake.
	msg, err := proto.ReadMsg(stream)
	if err != nil {
		slog.Error("read auth msg", "remote", remote, "err", err)
		return
	}
	if msg.Type != proto.TypeAuth {
		_ = proto.WriteMsg(stream, &proto.ControlMsg{Type: proto.TypeAuthFail, Error: "expected auth"})
		return
	}
	if !cfg.Dev {
		ok, err := cfg.TokenStore.Validate(msg.Token)
		if err != nil || !ok {
			_ = proto.WriteMsg(stream, &proto.ControlMsg{Type: proto.TypeAuthFail, Error: "invalid token"})
			slog.Warn("auth rejected", "remote", remote)
			return
		}
	}
	if err := proto.WriteMsg(stream, &proto.ControlMsg{Type: proto.TypeAuthOK}); err != nil {
		slog.Error("write auth_ok", "err", err)
		return
	}
	slog.Info("client authenticated", "remote", remote)

	// Handle registration messages until the control stream closes.
	for {
		msg, err := proto.ReadMsg(stream)
		if err != nil {
			slog.Debug("control stream closed", "remote", remote)
			return
		}
		switch msg.Type {
		case proto.TypeRegister:
			handleRegister(ctx, conn, stream, cfg, msg, remote)
		case proto.TypeUnregister:
			cfg.Registry.Unregister(msg.TunnelID)
			slog.Info("tunnel unregistered", "id", msg.TunnelID)
		default:
			slog.Warn("unexpected control msg", "type", msg.Type)
		}
	}
}

func handleRegister(ctx context.Context, conn quic.Connection, w interface{ Write([]byte) (int, error) }, cfg Config, msg *proto.ControlMsg, remote string) {
	switch msg.Proto {
	case "http":
		id, err := cfg.Registry.RegisterHTTP(msg.Name, conn)
		if err != nil {
			_ = proto.WriteMsg(w, &proto.ControlMsg{Type: proto.TypeError, Error: err.Error()})
			return
		}
		url := "https://" + cfg.Registry.LookupByID(id).Subdomain + "." + cfg.Domain
		_ = proto.WriteMsg(w, &proto.ControlMsg{Type: proto.TypeRegistered, TunnelID: id, URL: url})
		slog.Info("HTTP tunnel registered", "url", url, "remote", remote)

	case "tcp":
		id, port, err := cfg.Registry.RegisterTCP(conn)
		if err != nil {
			_ = proto.WriteMsg(w, &proto.ControlMsg{Type: proto.TypeError, Error: err.Error()})
			return
		}
		addr := fmt.Sprintf("%s:%d", cfg.Domain, port)
		_ = proto.WriteMsg(w, &proto.ControlMsg{Type: proto.TypeRegistered, TunnelID: id, Addr: addr})
		slog.Info("TCP tunnel registered", "port", port, "remote", remote)
		go serveTCPTunnel(ctx, conn, id, port, cfg.Registry)

	default:
		_ = proto.WriteMsg(w, &proto.ControlMsg{Type: proto.TypeError, Error: "unknown proto: " + msg.Proto})
	}
}

import (
	"context"

// TunnelSpec describes one --expose flag.
type TunnelSpec struct {
	LocalPort uint16
	Proto     string // "http" or "tcp"
	Name      string // optional subdomain name
}

// ParseTunnelSpec parses "3000:http", "3000:http:my-app", or "5432:tcp".
func ParseTunnelSpec(s string) (TunnelSpec, error) {
	parts := strings.SplitN(s, ":", 3)
	if len(parts) < 2 {
		return TunnelSpec{}, fmt.Errorf("invalid expose spec %q: want PORT:PROTO[:NAME]", s)
	}
	var port uint16
	if _, err := fmt.Sscanf(parts[0], "%d", &port); err != nil || port == 0 {
		return TunnelSpec{}, fmt.Errorf("invalid port in %q", s)
	}
	if parts[1] != "http" && parts[1] != "tcp" {
		return TunnelSpec{}, fmt.Errorf("unknown proto %q in %q", parts[1], s)
	}
	name := ""
	if len(parts) == 3 {
		name = parts[2]
	}
	return TunnelSpec{LocalPort: port, Proto: parts[1], Name: name}, nil
}

// Config is the client runtime configuration.
type Config struct {
	Server   string
	Token    string
	Tunnels  []TunnelSpec
	Insecure bool
}

// Connect dials the server and reconnects with exponential backoff until ctx is cancelled.
func Connect(ctx context.Context, cfg Config) {
	backoff := time.Second
	for {
		if err := connect(ctx, cfg); err != nil && ctx.Err() == nil {
			slog.Error("disconnected, reconnecting", "err", err, "in", backoff)
			select {
			case <-ctx.Done():
				return
			case <-time.After(backoff):
			}
			if backoff < 30*time.Second {
				backoff *= 2
			}
		}
		if ctx.Err() != nil {
			return
		}
	}
}

func connect(ctx context.Context, cfg Config) error {
	addr := cfg.Server
	if !strings.Contains(addr, ":") {
		addr += ":443"
	}
	tlsCfg := &tls.Config{
		InsecureSkipVerify: cfg.Insecure, //nolint:gosec // only set via --insecure in dev mode
		NextProtos:         []string{"rift-v1"},
	}
	quicCfg := &quic.Config{
		MaxIdleTimeout:  30 * time.Second,
		KeepAlivePeriod: 15 * time.Second,
		Allow0RTT:       true,
	}
	conn, err := quic.DialAddr(ctx, addr, tlsCfg, quicCfg)
	if err != nil {
		return fmt.Errorf("dial %s: %w", addr, err)
	}
	defer conn.CloseWithError(0, "bye")
	slog.Info("connected", "server", addr)

	ctrl, err := conn.OpenStreamSync(ctx)
	if err != nil {
		return fmt.Errorf("open control stream: %w", err)
	}

	if err := proto.WriteMsg(ctrl, &proto.ControlMsg{Type: proto.TypeAuth, Token: cfg.Token}); err != nil {
		return fmt.Errorf("send auth: %w", err)
	}
	resp, err := proto.ReadMsg(ctrl)
	if err != nil {
		return fmt.Errorf("read auth response: %w", err)
	}
	if resp.Type != proto.TypeAuthOK {
		return fmt.Errorf("auth rejected: %s", resp.Error)
	}
	slog.Info("authenticated")

	tunnelIDs := make(map[uint32]TunnelSpec)
	for _, spec := range cfg.Tunnels {
		if err := proto.WriteMsg(ctrl, &proto.ControlMsg{
			Type:  proto.TypeRegister,
			Port:  spec.LocalPort,
			Proto: spec.Proto,
			Name:  spec.Name,
		}); err != nil {
			return fmt.Errorf("send register: %w", err)
		}
		reg, err := proto.ReadMsg(ctrl)
		if err != nil {
			return fmt.Errorf("read register response: %w", err)
		}
		if reg.Type == proto.TypeError {
			slog.Error("tunnel registration failed", "err", reg.Error)
			continue
		}
		tunnelIDs[reg.TunnelID] = spec
		if reg.URL != "" {
			slog.Info("tunnel ready", "url", reg.URL, "local", fmt.Sprintf("localhost:%d", spec.LocalPort))
		} else {
			slog.Info("tunnel ready", "addr", reg.Addr, "local", fmt.Sprintf("localhost:%d", spec.LocalPort))
		}
	}

	return acceptDataStreams(ctx, conn, tunnelIDs)
}

// acceptDataStreams accepts server-opened QUIC streams, reads the tunnel header,
// dials the local service, and relays bidirectionally.
func acceptDataStreams(ctx context.Context, conn quic.Connection, tunnels map[uint32]TunnelSpec) error {
	for {
		stream, err := conn.AcceptStream(ctx)
		if err != nil {
			if ctx.Err() != nil {
				return nil
			}
			return fmt.Errorf("accept data stream: %w", err)
		}
		go handleDataStream(ctx, stream, tunnels)
	}
}

func handleDataStream(ctx context.Context, stream quic.Stream, tunnels map[uint32]TunnelSpec) {
	defer stream.Close()

	hdr, err := proto.ReadHeader(stream)
	if err != nil {
		slog.Error("read tunnel header", "err", err)
		return
	}
	spec, ok := tunnels[hdr.TunnelID]
	if !ok {
		slog.Error("unknown tunnel ID", "id", hdr.TunnelID)
		return
	}

	local, err := net.Dial("tcp", fmt.Sprintf("localhost:%d", spec.LocalPort))
	if err != nil {
		slog.Error("connect to local service", "port", spec.LocalPort, "err", err)
		return
	}
	defer local.Close()

	slog.Debug("relaying", "tunnel_id", hdr.TunnelID, "port", spec.LocalPort)
	relay.Relay(ctx, local, stream)
}

		return fmt.Errorf("HTTPS listen %s: %w", addr, err)
	}
	slog.Info("HTTPS listener started", "addr", addr)
	srv := &http.Server{
		Handler:      &httpHandler{registry: reg},
		ReadTimeout:  30 * time.Second,
		WriteTimeout: 30 * time.Second,
	}
	go func() {
		<-ctx.Done()
		shutCtx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
		defer cancel()
		_ = srv.Shutdown(shutCtx)
	}()
	if err := srv.Serve(ln); err != http.ErrServerClosed {
		return err
	}
	return nil
}

type httpHandler struct{ registry *TunnelRegistry }

func (h *httpHandler) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	host := r.Host
	if i := strings.IndexByte(host, ':'); i != -1 {
		host = host[:i]
	}
	subdomain := strings.SplitN(host, ".", 2)[0]
	tun := h.registry.LookupBySubdomain(subdomain)
	if tun == nil {
		http.Error(w, "tunnel not found", http.StatusNotFound)
		return
	}
	conn, ok := tun.Conn.(quic.Connection)
	if !ok {
		http.Error(w, "internal error", http.StatusInternalServerError)
		return
	}
	if strings.EqualFold(r.Header.Get("Upgrade"), "websocket") {
		proxyWebSocket(w, r, conn, tun.ID)
		return
	}
	rp := &httputil.ReverseProxy{
		Director: func(req *http.Request) {
			req.URL.Scheme = "http"
			req.URL.Host = r.Host
			req.Header.Set("X-Forwarded-For", r.RemoteAddr)
			req.Header.Set("X-Forwarded-Proto", "https")
			req.Header.Set("X-Forwarded-Host", r.Host)
		},
		Transport: &quicTransport{conn: conn, tunnelID: tun.ID},
	}
	rp.ServeHTTP(w, r)
}

type quicTransport struct {
	conn     quic.Connection
	tunnelID uint32
}

func (t *quicTransport) RoundTrip(req *http.Request) (*http.Response, error) {
	stream, err := t.conn.OpenStreamSync(req.Context())
	if err != nil {
		return nil, fmt.Errorf("open QUIC stream: %w", err)
	}
	if err := proto.WriteHeader(stream, proto.TunnelHeader{TunnelID: t.tunnelID}); err != nil {
		stream.Close()
		return nil, fmt.Errorf("write tunnel header: %w", err)
	}
	if err := req.Write(stream); err != nil {
		stream.Close()
		return nil, fmt.Errorf("write HTTP request to stream: %w", err)
	}
	resp, err := http.ReadResponse(bufio.NewReader(stream.(io.Reader)), req)
	if err != nil {
		stream.Close()
		return nil, fmt.Errorf("read HTTP response from stream: %w", err)
	}
	return resp, nil
}

func proxyWebSocket(w http.ResponseWriter, r *http.Request, conn quic.Connection, tunnelID uint32) {
	stream, err := conn.OpenStreamSync(r.Context())
	if err != nil {
		http.Error(w, "bad gateway", http.StatusBadGateway)
		return
	}
	if err := proto.WriteHeader(stream, proto.TunnelHeader{TunnelID: tunnelID}); err != nil {
		stream.Close()
		http.Error(w, "bad gateway", http.StatusBadGateway)
		return
	}
	if err := r.Write(stream); err != nil {
		stream.Close()
		http.Error(w, "bad gateway", http.StatusBadGateway)
		return
	}
	hj, ok := w.(http.Hijacker)
	if !ok {
		stream.Close()
		http.Error(w, "websocket not supported", http.StatusInternalServerError)
		return
	}
	clientConn, _, err := hj.Hijack()
	if err != nil {
		stream.Close()
		return
	}
	defer clientConn.Close()
	relay.Relay(r.Context(), clientConn, stream.(io.ReadWriteCloser))
}

// serveTCPTunnel binds port and forwards each visitor TCP connection through a QUIC stream.
func serveTCPTunnel(ctx context.Context, conn quic.Connection, id uint32, port uint16, reg *TunnelRegistry) {
	ln, err := net.Listen("tcp", fmt.Sprintf(":%d", port))
	if err != nil {
		slog.Error("TCP tunnel listen failed", "port", port, "err", err)
		reg.Unregister(id)
		return
	}
	slog.Info("TCP tunnel listener", "port", port)
	go func() {
		<-ctx.Done()
		ln.Close()
	}()
	defer func() {
		ln.Close()
		reg.Unregister(id)
	}()
	for {
		visitor, err := ln.Accept()
		if err != nil {
			if ctx.Err() != nil {
				return
			}
			slog.Error("TCP accept", "port", port, "err", err)
			return
		}
		go forwardTCPVisitor(ctx, visitor, conn, id)
	}
}

func forwardTCPVisitor(ctx context.Context, visitor net.Conn, conn quic.Connection, tunnelID uint32) {
	defer visitor.Close()
	stream, err := conn.OpenStreamSync(ctx)
	if err != nil {
		slog.Error("open QUIC stream for TCP", "err", err)
		return
	}
	defer stream.Close()
	if err := proto.WriteHeader(stream, proto.TunnelHeader{TunnelID: tunnelID}); err != nil {
		slog.Error("write TCP tunnel header", "err", err)
		return
	}
	relay.Relay(ctx, visitor, stream.(io.ReadWriteCloser))
}

func runServer(args []string) {
	fs := flag.NewFlagSet("server", flag.ExitOnError)
	domain    := fs.String("domain",     "tunnel.localhost", "Base domain for HTTP tunnels")
	listen    := fs.String("listen",     ":443",             "Listen address (QUIC=UDP, HTTPS=TCP share this port)")
	dev       := fs.Bool  ("dev",        false,              "Dev mode: self-signed cert, no token auth")
	tokenFile := fs.String("token-file", "/etc/rift/tokens", "Server token file path")
	certFile  := fs.String("cert",       "",                 "TLS cert PEM file (for pre-provisioned wildcard cert)")
	keyFile   := fs.String("key",        "",                 "TLS key PEM file (required with --cert)")
	addToken  := fs.String("add-token",  "",                 "Add a named token entry and exit (e.g. --add-token alice)")
	_ = fs.Parse(args)

	if *addToken != "" {
		tok, err := auth.GenerateToken()
		if err != nil {
			fmt.Fprintf(os.Stderr, "error: %v\n", err)
			os.Exit(1)
		}
		if err := auth.NewServerTokenStore(*tokenFile).Add(*addToken, tok); err != nil {
			fmt.Fprintf(os.Stderr, "error: %v\n", err)
			os.Exit(1)
		}
		fmt.Printf("Token for %q:\n%s\n", *addToken, tok)
		return
	}

	var tlsCfg *tls.Config
	switch {
	case *dev:
		var err error
		tlsCfg, err = server.DevTLSConfig(*domain)
		if err != nil {
			slog.Error("dev TLS", "err", err)
			os.Exit(1)
		}
		slog.Info("dev mode active — self-signed cert, auth disabled")
	case *certFile != "":
		cert, err := tls.LoadX509KeyPair(*certFile, *keyFile)
		if err != nil {
			slog.Error("load TLS cert/key", "err", err)
			os.Exit(1)
		}
		tlsCfg = &tls.Config{Certificates: []tls.Certificate{cert}}
	default:
		tlsCfg = server.ProdTLSConfig(*domain, "/etc/rift/certs")
	}

	reg := server.NewTunnelRegistry()
	cfg := server.Config{
		ListenAddr: *listen,
		Domain:     *domain,
		Dev:        *dev,
		TLS:        tlsCfg,
		Registry:   reg,
	}
	if !*dev {
		cfg.TokenStore = auth.NewServerTokenStore(*tokenFile)
	}

	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	// TCP 443 = HTTPS, UDP 443 = QUIC — no conflict.
	go func() {
		if err := server.ServeHTTP(ctx, *listen, tlsCfg, reg); err != nil {
			slog.Error("HTTPS server error", "err", err)
		}
	}()

	if err := server.Run(ctx, cfg); err != nil {
		slog.Error("server error", "err", err)
		os.Exit(1)
	}
}

	_ = fs.Parse(args)

		fmt.Fprintln(os.Stderr, "error: --server is required")

			os.Exit(1)

	token := *tokenFlag

		home, _ := os.UserHomeDir()
